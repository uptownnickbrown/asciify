Add your Twitter app credentials to config.json with this structure:

{
  twitter_creds: {
    consumer_key: 'YOURKEY',
    consumer_secret: 'YOURSECRET',
    access_token: 'YOURTOKEN',
    access_token_secret: 'YOURTOKENSECRET'
  }
}
